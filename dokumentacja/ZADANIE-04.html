Zadanie IV – Jenkins Pipeline

Pipeline Script: 

node {
 stage('GIT Checkout') {
  git branch: 'master', url: 'https://github.com/tklepacki/aa-days.git'
 }
 stage('Test Run') {
  if (isUnix()) {
   sh "./mvnw clean install"
  } else {
   bat("mvnw clean install")
  }
 }
 stage('Results') {
  step([$class: 'Publisher', reportFilenamePattern: '**/testng-results.xml'])
 }
}
